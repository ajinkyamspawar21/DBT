1. Find sum and avg of sal and netsal for each job
   net sal=sal+comm
---------------------------------------------------------------------------------------------------------------------------
select  job, sum(sal),avg(sal),sum(sal+comm) "net_sal"
from emp
group by job;


---------------------------------------------------------------------------------------------------------------------------
2. Find how many analysts are there in each department
---------------------------------------------------------------------------------------------------------------------------
select job, count(job)
from emp
group by job
having job='ANALYST';
---------------------------------------------------------------------------------------------------------------------------
3. Find sum , average of salary for all employees whose sal> 1500 departmentwise
---------------------------------------------------------------------------------------------------------------------------
select deptno, sum(sal), avg(sal)
from emp
group by deptno
having sum(sal)>1500;
---------------------------------------------------------------------------------------------------------------------------
4. Find(COUNT) all salesman also by (MANAGER) in each department
---------------------------------------------------------------------------------------------------------------------------
select deptno,count(*)
from emp 
where job='MANAGER'
group by deptno;

---------------------------------------------------------------------------------------------------------------------------
5. to find how many employees are working under same manager;
---------------------------------------------------------------------------------------------------------------------------

group by mgr;

6. To find max, min and sum of salary for all employees whose sal > 1500 and < 3000
---------------------------------------------------------------------------------------------------------------------------
select min(sal),max(sal),sum(sal)
from emp
having sum(sal>1500)and (sal<3000);
